CONFIGURE CONTROLFILE AUTOBACKUP ON;
CONFIGURE RETENTION POLICY TO RECOVERY WINDOW OF 28 DAYS;
CONFIGURE ARCHIVELOG DELETION POLICY TO APPLIED ON ALL STANDBY BACKED UP 1 TIMES TO ‘SBT_TAPE’;




run {
	ALLOCATE CHANNEL CH01 CONNECT 'SYS/PASSWD@TNS_INST1' TYPE 'SBT_TAPE';
	ALLOCATE CHANNEL CH02 CONNECT 'SYS/PASSWD@TNS_INST1' TYPE 'SBT_TAPE';
	ALLOCATE CHANNEL CH03 CONNECT 'SYS/PASSWD@TNS_INST2' TYPE 'SBT_TAPE';
	ALLOCATE CHANNEL CH04 CONNECT 'SYS/PASSWD@TNS_INST2' TYPE 'SBT_TAPE';
		
	BACKUP AS BACKUPSET INCREMENTAL LEVEL 0 FORMAT '%d_%U' DATABASE FILESPERSET 4 MAXSETSIZE 40 G TAG LEVEL0;
		
	BACKUP AS COMPRESSED BACKUPSET CURRENT CONTROLFILE TAG CTRL;
	
	SQL 'ALTER SYSTEM ARCHIVE LOG CURRENT';
	BACKUP AS BACKUPSET NOT BACKED UP ARCHIVELOG ALL FILESPERSET 4 TAG ARCH ;
	
	DELETE NOPROMPT OBSOLETE RECOVERY WINDOW OF 28 DAYS;
	DELETE NOPROMPT ARCHIVELOG UNTIL TIME 'SYSDATE-3';
	
	CROSSCHECK BACKUP;
	CROSSCHECK COPY;
	DELETE NOPROMPT EXPIRED BACKUP;

	RELEASE CHANNEL CH01;
	RELEASE CHANNEL CH02;
	RELEASE CHANNEL CH03;
	RELEASE CHANNEL CH04;

	}
EXIT;
