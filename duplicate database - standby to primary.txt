1.	shutdown 1 node
------------------------------------------------------------------

2.	insert into listener.ora --> restart listener
------------------------------------------------------------------
vi $ORACLE_HOME/network/admin/listener.ora
SID_LIST_LISTENER =
(SID_LIST =
        (SID_DESC =
        (GLOBAL_DBNAME = cobr14dr)
        (ORACLE_HOME = /u01/app/oracle/product/12.1.0/dbhome_1)
        (SID_NAME = twrres)
    )
)
                                
-- restart listener
lsnrctl stop                          
lsnrctl start
